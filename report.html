def performance_metric2(y_true, y_predict):
    """"""
    SSE = ESS = 0.0
    avg_y_ = sum(y_true)/len(y_true)
    for i in range(0,len(y_predict)):
        S = pow(y_true[i] - avg_y_, 2)
        SSE = SSE + S
    for i in range(0, len(y_true)):
        E = pow(y_true[i] - y_predict[i], 2)
        ESS = ESS + E
    score = 1- ESS / SSE

    return  score
